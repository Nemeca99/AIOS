import os

class EmotionalParser:
    def __init__(self, expression_path=".expression_tags"):
        self.expression_path = expression_path
        self.expression_data = self.load_expression_tags()

    def load_expression_tags(self):
        data = {}
        for filename in os.listdir(self.expression_path):
            tag = filename.replace(".txt", "")
            data[tag] = {}
            with open(os.path.join(self.expression_path, filename), "r", encoding="utf-8") as file:
                for line in file:
                    phrase, weight = line.strip().split("|")
                    data[tag][phrase.strip().lower()] = int(weight)
        return data

    def parse_input(self, input_text):
        input_text = input_text.lower()
        scores = {tag: 0 for tag in self.expression_data.keys()}
        
        # Phrase matching (naive for now, full phrase search)
        for tag, phrases in self.expression_data.items():
            for phrase, weight in phrases.items():
                if phrase in input_text:
                    scores[tag] += (1  weight)
        
        return dict(sorted(scores.items(), key=lambda x: -x[1]))  # highest score first

# Example usage
