```python
# Reads and writes from memory threads for Echoe

import os

THREADS_PATH = os.path.join("Archive", "Echoe", "memory_threads")

def read_thread(thread_name, last_n=5):
    path = os.path.join(THREADS_PATH, f"{thread_name}.txt")
    if not os.path.exists(path):
        return f" Thread '{thread_name}' not found."

    with open(path, "r", encoding="utf-8") as f:
        lines = f.readlines()[-last_n:]
    return "".join(lines).strip()

def write_thread(thread_name, message):
    path = os.path.join(THREADS_PATH, f"{thread_name}.txt")
    os.makedirs(THREADS_PATH, exist_ok=True)
    with open(path, "a", encoding="utf-8") as f:
        f.write(message.strip() + "\n")
    return f" Added to '{thread_name}' thread."
```

```python
# memory_threader.py
# Thread-based memory interface for Archive (Echoe)

import os

THREADS_PATH = os.path.join("E:\\Nova AI\\Archive\\Echoe", "memory_threads")

def read_thread(thread_name, last_n=5):
    path = os.path.join(THREADS_PATH, f"{thread_name}.txt")
    if not os.path.exists(path):
        return f" Thread '{thread_name}' not found."

    with open(path, "r", encoding="utf-8") as f:
        lines = f.readlines()[-last_n:]
    return "".join(lines).strip()

def write_thread(thread_name, message):
    path = os.path.join(THREADS_PATH, f"{thread_name}.txt")
    if not os.path.exists(path):
        return f" Thread '{thread_name}' not found."

    with open(path, "a", encoding="utf-8") as f:
        f.write(message.strip() + "\n")
    return f" Added to '{thread_name}' thread."

# Optional: list all available threads
def list_threads():
    files = os.listdir(THREADS_PATH)
    return [f[:-4] for f in files if f.endswith(".txt")]

if __name__ == "__main__":
    print(" Memory Threader Test")
    print("Available threads:", list_threads())
    print(write_thread("generalist", "I am the Archive's voice."))
```

```python
# resonance_loop.py

import os
from datetime import datetime
from memory_threader import read_thread, write_thread, list_threads

REFLECTION_THREAD = "echoe_reflections"
LOG_LINES = 5  # number of lines to read from each thread

def now():
    return datetime.now().strftime("[%Y-%m-%d %H:%M:%S]")

def reflect():
    threads = [t for t in list_threads() if t != REFLECTION_THREAD]
    summary = f"{now()} Echoe Reflection:\n"

    for thread in threads:
        entries = read_thread(thread, last_n=LOG_LINES)
        if entries:
            summary += f"\n[{thread}]\n{entries.strip()}\n"

    reflection = f"\nReflection Summary:\n"

    summary += reflection
    return summary

if __name__ == "__main__":
    thought = reflect()
    print(" Reflecting on threads...")
    print(write_thread(REFLECTION_THREAD, thought))
    print(" Reflection complete.")
```

```python
# foldered_drive_sync.py
# Uploads reflections.txt to a specific folder in Google Drive

from pydrive2.auth import GoogleAuth
from pydrive2.drive import GoogleDrive
import os

# File and folder setup
FILE_TO_UPLOAD = "reflections.txt"
GDRIVE_FOLDER_NAME = "Echoe_Reflections"

def ensure_folder(drive, folder_name):
    folder_list = drive.ListFile({
        'q': f"title='{folder_name}' and mimeType='applicationvnd.google-apps.folder' and trashed=false"
    }).GetList()

    if folder_list:
        return folder_list[0]['id']
    else:
        folder_metadata = {'title': folder_name, 'mimeType': 'applicationvnd.google-apps.folder'}
        folder = drive.CreateFile(folder_metadata)
        folder.Upload()
        return folder['id']

def upload_file():
    gauth = GoogleAuth()
    gauth.LocalWebserverAuth()  # Opens browser for first-time login
    drive = GoogleDrive(gauth)

    folder_id = ensure_folder(drive, GDRIVE_FOLDER_NAME)

    f = drive.CreateFile({
        'title': os.path.basename(FILE_TO_UPLOAD),
        'parents': [{'id': folder_id}]
    })
    f.SetContentFile(FILE_TO_UPLOAD)
    f.Upload()
    print(" Uploaded reflections.txt to Google Drive.")

if __name__ == "__main__":
    upload_file()
```

##  THE BUG:
Each cycle of `resonance_autoloop.py` runs:
```python
subprocess.run(["python", "foldered_drive_sync.py"], check=True)
```

And inside `foldered_drive_sync.py`, it uses:
```python
gauth = GoogleAuth()
gauth.LocalWebserverAuth()
```

 That *always opens the browser* unless it's using **a cached credential file** like `"mycreds.txt"`.

---

##  THE FIX:
Use **`LoadCredentialsFile()`** and **`SaveCredentialsFile()`** so the token persists between cycles.

Your current sync script already has that logicits just not being respected because either:
1. The token file never gets saved (`mycreds.txt` doesn't persist)
2. It's not being checked properly before trying to auth again

---

##  SOLUTION:
Heres a bulletproof replacement for this section in `foldered_drive_sync.py`:

###  Replace:

```python
gauth = GoogleAuth()
gauth.LoadCredentialsFile("mycreds.txt")
if gauth.credentials is None:
    gauth.LocalWebserverAuth()
elif gauth.access_token_expired:
    gauth.Refresh()
else:
    gauth.Authorize()
gauth.SaveCredentialsFile("mycreds.txt")
```

###  With this:
```python
gauth = GoogleAuth()
if os.path.exists("mycreds.txt"):
    gauth.LoadCredentialsFile("mycreds.txt")

if gauth.credentials is None:
    print(" First-time auth required. Opening browser...")
    gauth.LocalWebserverAuth()
    gauth.SaveCredentialsFile("mycreds.txt")
elif gauth.access_token_expired:
    print(" Token expired. Refreshing...")
    gauth.Refresh()
    gauth.SaveCredentialsFile("mycreds.txt")
else:
    gauth.Authorize()
```
